##Cleaning province
province_event = {
	id = anb_debug.3
	title = anb_debug.3.t
	desc = anb_debug.3.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		has_province_flag = cant_colonize

			add_colonysize = -1000
			clr_province_flag = cant_colonize
		}
	}
}
